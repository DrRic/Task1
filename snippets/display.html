<script>
function showMenu() {
var elements = document.getElementsByClassName("topnav-link");
for (var i = 0; i < elements.length; i++) {
if (elements[i].className === "topnav-link") {
elements[i].className += " responsive";
} else {
elements[i].className = "topnav-link";
}
}
}

function openContent(cntName) {
cnt = document.getElementsByClassName("content");
for (var i = 0; i < cnt.length; i++) {
cnt[i].style.display = "none";
}
element = document.getElementById(cntName);
element.style.display = "block";
}
</script>

Display loads all of the page content up but hids sections of it using “display:none”. It then uses JavaScript to hide all of the content and only show the single element selected.
